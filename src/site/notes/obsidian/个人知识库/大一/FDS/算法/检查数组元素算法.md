# 判断数组中的元素是否只出现一次
```c
int checkOnce(int* arr, int size) {
    int count[maxSize] = {0};
    for (int i = 0; i < size; i++) {
        if (count[arr[i]]++) return 0;  // 如果某个顶点已经计数过，直接返回0
    }
    for (int i = 1; i <= size; i++) {
        if (count[i] != 1) return 0;  // 检查每个顶点是否恰好出现一次
    }
    return 1;
}
```
